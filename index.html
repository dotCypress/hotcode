Practice

var hotcode = new Mongo().getDB('hotcode');
var albumImages = hotcode.albums.find().toArray().
  reduce(function(a, b) {
    return a.concat(b.images);
  }, []);
var selector = { _id: { $not: { $in: albumImages }}};
hotcode.images.remove(selector);
